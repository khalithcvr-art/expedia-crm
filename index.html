🗄 Supabase RBAC SQL Reference

Run these SQL scripts in Supabase SQL Editor to set up the full RBAC system with RLS policies, functions, and triggers.

1. Create Tables 
-- ROLES
CREATE TABLE IF NOT EXISTS roles (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL UNIQUE,
  level INT NOT NULL DEFAULT 99,
  created_at TIMESTAMPTZ DEFAULT now()
);
INSERT INTO roles (name, level) VALUES
  ('Super Admin', 1),
  ('Admin', 2),
  ('Manager', 3),
  ('Pro User', 4),
  ('Staff', 5)
ON CONFLICT (name) DO NOTHING;

-- PERMISSIONS
CREATE TABLE IF NOT EXISTS permissions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  module TEXT NOT NULL,
  action TEXT NOT NULL,
  UNIQUE(module, action)
);
INSERT INTO permissions (module, action) VALUES
  ('clients','view'),('clients','create'),('clients','edit'),('clients','delete'),('clients','export'),
  ('customers','view'),('customers','create'),('customers','edit'),('customers','delete'),('customers','export'),
  ('invoices','view'),('invoices','create'),('invoices','edit'),('invoices','delete'),('invoices','approve'),('invoices','export'),
  ('quotations','view'),('quotations','create'),('quotations','edit'),('quotations','delete'),('quotations','approve'),('quotations','export'),
  ('employees','view'),('employees','create'),('employees','edit'),('employees','delete'),('employees','export'),
  ('tasks','view'),('tasks','create'),('tasks','edit'),('tasks','delete'),
  ('transactions','view'),('transactions','create'),('transactions','edit'),('transactions','delete'),('transactions','export'),
  ('documents','view'),('documents','create'),('documents','delete'),
  ('users','view'),('users','create'),('users','edit'),('users','delete')
ON CONFLICT (module, action) DO NOTHING;

-- ROLE_PERMISSIONS
CREATE TABLE IF NOT EXISTS role_permissions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  role_id UUID REFERENCES roles(id) ON DELETE CASCADE,
  permission_id UUID REFERENCES permissions(id) ON DELETE CASCADE,
  UNIQUE(role_id, permission_id)
);

-- COMPANIES (if not exists)
CREATE TABLE IF NOT EXISTS companies (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- USER_COMPANIES
CREATE TABLE IF NOT EXISTS user_companies (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  company_id UUID REFERENCES companies(id) ON DELETE CASCADE,
  UNIQUE(user_id, company_id)
);

-- USER_PROFILES
CREATE TABLE IF NOT EXISTS user_profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id),
  name TEXT NOT NULL,
  role_id UUID REFERENCES roles(id),
  data_scope TEXT DEFAULT 'own' CHECK (data_scope IN ('own','assigned','company')),
  status TEXT DEFAULT 'active' CHECK (status IN ('active','inactive')),
  created_at TIMESTAMPTZ DEFAULT now()
);

-- AUDIT_LOGS
CREATE TABLE IF NOT EXISTS audit_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID,
  action TEXT NOT NULL,
  module TEXT NOT NULL,
  record_id TEXT,
  details JSONB,
  created_at TIMESTAMPTZ DEFAULT now()
);
2. RBAC Functions 
-- GET USER ROLE
CREATE OR REPLACE FUNCTION get_user_role(p_user_id UUID)
RETURNS TABLE(role_name TEXT, role_level INT) AS $$
  SELECT r.name, r.level
  FROM user_profiles up
  JOIN roles r ON r.id = up.role_id
  WHERE up.id = p_user_id;
$$ LANGUAGE sql SECURITY DEFINER STABLE;

-- HAS PERMISSION CHECK
CREATE OR REPLACE FUNCTION has_permission(
  p_user_id UUID, p_module TEXT, p_action TEXT
) RETURNS BOOLEAN AS $$
  SELECT EXISTS (
    SELECT 1
    FROM user_profiles up
    JOIN role_permissions rp ON rp.role_id = up.role_id
    JOIN permissions p ON p.id = rp.permission_id
    WHERE up.id = p_user_id
      AND p.module = p_module
      AND p.action = p_action
      AND up.status = 'active'
  );
$$ LANGUAGE sql SECURITY DEFINER STABLE;

-- USER BELONGS TO COMPANY
CREATE OR REPLACE FUNCTION user_in_company(
  p_user_id UUID, p_company_id UUID
) RETURNS BOOLEAN AS $$
  SELECT EXISTS (
    SELECT 1 FROM user_companies
    WHERE user_id = p_user_id AND company_id = p_company_id
  );
$$ LANGUAGE sql SECURITY DEFINER STABLE;

-- PREVENT ROLE ESCALATION
CREATE OR REPLACE FUNCTION prevent_role_escalation()
RETURNS TRIGGER AS $$
DECLARE
  actor_level INT;
  target_level INT;
BEGIN
  SELECT r.level INTO actor_level
  FROM user_profiles up JOIN roles r ON r.id = up.role_id
  WHERE up.id = auth.uid();

  SELECT level INTO target_level FROM roles WHERE id = NEW.role_id;

  IF target_level < actor_level THEN
    RAISE EXCEPTION 'Cannot assign a role higher than your own';
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER trg_prevent_role_escalation
  BEFORE INSERT OR UPDATE ON user_profiles
  FOR EACH ROW EXECUTE FUNCTION prevent_role_escalation();

-- APPROVAL SYSTEM: only Manager+ can approve
CREATE OR REPLACE FUNCTION check_approval_permission()
RETURNS TRIGGER AS $$
DECLARE
  user_level INT;
BEGIN
  IF NEW.status = 'approved' AND (OLD.status IS DISTINCT FROM 'approved') THEN
    SELECT r.level INTO user_level
    FROM user_profiles up JOIN roles r ON r.id = up.role_id
    WHERE up.id = auth.uid();

    IF user_level IS NULL OR user_level > 3 THEN
      RAISE EXCEPTION 'Only Manager or higher can approve';
    END IF;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Apply to invoices and quotations
CREATE TRIGGER trg_invoice_approval
  BEFORE UPDATE ON invoices FOR EACH ROW
  EXECUTE FUNCTION check_approval_permission();

CREATE TRIGGER trg_quotation_approval
  BEFORE UPDATE ON quotations FOR EACH ROW
  EXECUTE FUNCTION check_approval_permission();
3. RLS Policies 
-- Enable RLS on all tables
ALTER TABLE roles ENABLE ROW LEVEL SECURITY;
ALTER TABLE permissions ENABLE ROW LEVEL SECURITY;
ALTER TABLE role_permissions ENABLE ROW LEVEL SECURITY;
ALTER TABLE user_profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE user_companies ENABLE ROW LEVEL SECURITY;
ALTER TABLE audit_logs ENABLE ROW LEVEL SECURITY;

-- USER_PROFILES: users read own profile
CREATE POLICY "Users read own profile"
  ON user_profiles FOR SELECT
  USING (auth.uid() = id);

-- USER_PROFILES: Super Admin manages all
CREATE POLICY "Super Admin manages profiles"
  ON user_profiles FOR ALL
  USING (
    (SELECT r.level FROM user_profiles up
     JOIN roles r ON r.id = up.role_id
     WHERE up.id = auth.uid()) = 1
  );

-- EXAMPLE: INVOICES RLS
CREATE POLICY "Invoice SELECT"
  ON invoices FOR SELECT USING (
    user_in_company(auth.uid(), company_id)
    AND has_permission(auth.uid(), 'invoices', 'view')
  );
CREATE POLICY "Invoice INSERT"
  ON invoices FOR INSERT WITH CHECK (
    user_in_company(auth.uid(), company_id)
    AND has_permission(auth.uid(), 'invoices', 'create')
  );
CREATE POLICY "Invoice UPDATE"
  ON invoices FOR UPDATE USING (
    user_in_company(auth.uid(), company_id)
    AND has_permission(auth.uid(), 'invoices', 'edit')
  );
CREATE POLICY "Invoice DELETE"
  ON invoices FOR DELETE USING (
    user_in_company(auth.uid(), company_id)
    AND has_permission(auth.uid(), 'invoices', 'delete')
  );

-- ROLES: readable by authenticated users
CREATE POLICY "Roles readable"
  ON roles FOR SELECT
  USING (auth.role() = 'authenticated');

-- PERMISSIONS: readable by authenticated users
CREATE POLICY "Permissions readable"
  ON permissions FOR SELECT
  USING (auth.role() = 'authenticated');

-- ROLE_PERMISSIONS: readable, Super Admin writable
CREATE POLICY "Role perms readable"
  ON role_permissions FOR SELECT
  USING (auth.role() = 'authenticated');
CREATE POLICY "Super Admin manages role perms"
  ON role_permissions FOR ALL
  USING (
    (SELECT r.level FROM user_profiles up
     JOIN roles r ON r.id = up.role_id
     WHERE up.id = auth.uid()) = 1
  );

-- AUDIT_LOGS: users see own logs, Admin+ see all
CREATE POLICY "Users see own audit logs"
  ON audit_logs FOR SELECT
  USING (
    user_id = auth.uid()
    OR (SELECT r.level FROM user_profiles up
        JOIN roles r ON r.id = up.role_id
        WHERE up.id = auth.uid()) <= 2
  );

-- Only Super Admin can delete users
CREATE POLICY "Only Super Admin deletes users"
  ON user_profiles FOR DELETE USING (
    (SELECT r.level FROM user_profiles up
     JOIN roles r ON r.id = up.role_id
     WHERE up.id = auth.uid()) = 1
  );

-- Admin cannot modify Super Admin
CREATE POLICY "Admin cannot modify Super Admin"
  ON user_profiles FOR UPDATE USING (
    -- Target is not Super Admin OR actor IS Super Admin
    (SELECT r.level FROM roles r WHERE r.id = user_profiles.role_id) > 1
    OR (SELECT r.level FROM user_profiles up
        JOIN roles r ON r.id = up.role_id
        WHERE up.id = auth.uid()) = 1
  );
4. Audit Log Trigger 
-- Generic audit trigger function
CREATE OR REPLACE FUNCTION log_audit()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO audit_logs (user_id, action, module, record_id, details)
  VALUES (
    auth.uid(),
    TG_OP,
    TG_TABLE_NAME,
    COALESCE(NEW.id::text, OLD.id::text),
    jsonb_build_object(
      'operation', TG_OP,
      'table', TG_TABLE_NAME,
      'old', CASE WHEN TG_OP = 'DELETE' THEN row_to_json(OLD) ELSE NULL END,
      'new', CASE WHEN TG_OP != 'DELETE' THEN row_to_json(NEW) ELSE NULL END
    )
  );
  RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Apply to all CRM tables
CREATE TRIGGER audit_clients AFTER INSERT OR UPDATE OR DELETE ON clients
  FOR EACH ROW EXECUTE FUNCTION log_audit();
CREATE TRIGGER audit_customers AFTER INSERT OR UPDATE OR DELETE ON customers
  FOR EACH ROW EXECUTE FUNCTION log_audit();
CREATE TRIGGER audit_invoices AFTER INSERT OR UPDATE OR DELETE ON invoices
  FOR EACH ROW EXECUTE FUNCTION log_audit();
CREATE TRIGGER audit_quotations AFTER INSERT OR UPDATE OR DELETE ON quotations
  FOR EACH ROW EXECUTE FUNCTION log_audit();
CREATE TRIGGER audit_employees AFTER INSERT OR UPDATE OR DELETE ON employees
  FOR EACH ROW EXECUTE FUNCTION log_audit();
CREATE TRIGGER audit_tasks AFTER INSERT OR UPDATE OR DELETE ON tasks
  FOR EACH ROW EXECUTE FUNCTION log_audit();
CREATE TRIGGER audit_transactions AFTER INSERT OR UPDATE OR DELETE ON transactions
  FOR EACH ROW EXECUTE FUNCTION log_audit();
